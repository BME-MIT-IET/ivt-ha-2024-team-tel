27 def kth_to_last_dict(head, k):
28 """
29 This is a brute force method where we keep a dict the size of the list
30 Then we check it for the value we need. If the key is not in the dict,
31 our and statement will short circuit and return False
32 """
33 if not (head and k > -1):
34 return False
35 d = dict()
36 count = 0
37 while head:
38 d[count] = head
39 head = head.next
40 count += 1
41 return len(d)-k in d and d[len(d)-k]

44 def kth_to_last(head, k):
45 """
46 This is an optimal method using iteration.
47 We move p1 k steps ahead into the list.
48 Then we move p1 and p2 together until p1 hits the end.
49 """
50 if not (head or k > -1):
51 return False
52 p1 = head
53 p2 = head
54 for i in range(1, k+1):
55 if p1 is None:
56 # Went too far, k is not valid
57 raise IndexError
58 p1 = p1.next
59 while p1:
60 p1 = p1.next
61 p2 = p2.next
