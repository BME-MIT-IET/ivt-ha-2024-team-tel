27 def get_primes(n):
28 """Return list of all primes less than n,
29 Using sieve of Eratosthenes.
30 """
31 if n <= 0:
32 raise ValueError("'n' must be a positive integer.")
33 # If x is even, exclude x from list (-1):
34 sieve_size = (n // 2 - 1) if n % 2 == 0 else (n // 2)
35 sieve = [True for _ in range(sieve_size)] # Sieve
36 primes = [] # List of Primes
37 if n >= 2:
38 primes.append(2) # 2 is prime by default
39 for i in range(sieve_size):
40 if sieve[i]:
41 value_at_i = i*2 + 3
42 primes.append(value_at_i)
43 for j in range(i, sieve_size, value_at_i):
44 sieve[j] = False
45 return primes
