97 def solve_sat(formula):
98 """
99 Solves the 2-SAT problem
100 """
101 graph = build_graph(formula)
102 vertex_scc = scc(graph)
103  
104 for (var, _) in graph:
105 if vertex_scc[(var, False)] == vertex_scc[(var, True)]:
106 return None # The formula is contradictory
107  
108 comp_repr = {} # An arbitrary representant from each component
109  
110 for vertex in graph:
111 if not vertex_scc[vertex] in comp_repr:
112 comp_repr[vertex_scc[vertex]] = vertex
113  
114 comp_value = {} # True/False value for each strongly connected component
115 components = sorted(vertex_scc.values())
116  
117 for comp in components:
118 if comp not in comp_value:
119 comp_value[comp] = False
120  
121 (lit, neg) = comp_repr[comp]
122 comp_value[vertex_scc[(lit, not neg)]] = True
123  
124 value = {var: comp_value[vertex_scc[(var, False)]] for (var, _) in graph}
125  
126 return value
