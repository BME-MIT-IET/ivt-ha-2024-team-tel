14 - WHITE: has not been visited yet

22 def is_in_cycle(graph, traversal_states, vertex):
23 """
24 Determines if the given vertex is in a cycle.
25  
26 :param: traversal_states: for each vertex, the state it is in
27 """
28 if traversal_states[vertex] == TraversalState.GRAY:
29 return True
30 traversal_states[vertex] = TraversalState.GRAY
31 for neighbor in graph[vertex]:
32 if is_in_cycle(graph, traversal_states, neighbor):
33 return True
34 traversal_states[vertex] = TraversalState.BLACK
35 return False
