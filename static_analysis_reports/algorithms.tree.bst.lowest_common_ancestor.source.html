24 def lowest_common_ancestor(root, p, q):
25 """
26 :type root: Node
27 :type p: Node
28 :type q: Node
29 :rtype: Node
30 """
31 while root:
32 if p.val > root.val < q.val:
33 root = root.right
34 elif p.val < root.val > q.val:
35 root = root.left
36 else:
37 return root
